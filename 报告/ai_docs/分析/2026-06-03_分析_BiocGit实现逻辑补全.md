// 从版本号中提取次版本号（如 "1.50.5" → 50）
static int ParseMinorVersion(const std::wstring& version) {
    size_t first_dot = version.find(L'.');
    if (first_dot == std::wstring::npos) return -1;
    size_t second_dot = version.find(L'.', first_dot + 1);
    std::wstring minor_str;
    if (second_dot != std::wstring::npos) {
        minor_str = version.substr(first_dot + 1, second_dot - first_dot - 1);
    } else {
        minor_str = version.substr(first_dot + 1);
    }
    try {
        return std::stoi(minor_str);
    } catch (...) {
        return -1;
    }
}

// 根据包版本号推断 Bioconductor 版本
// 规则：Bioconductor 版本 = 3 + (包次版本号 / 2)
// 例如：1.50.5 → 次版本号 50 → 3 + 50/2 = 28 → 3.28（但实际是 3.18）
// 修正规则：Bioconductor 版本 = 3 + (包次版本号 - 1) / 2
// 例如：1.50.5 → 次版本号 50 → 3 + (50-1)/2 = 3 + 24 = 27 → 3.27（仍不对）
// 实际映射：1.50.x → 3.18, 1.52.x → 3.19, 1.54.x → 3.20
// 正确公式：Bioconductor 版本 = 3 + (包次版本号 - 34) / 2
// 例如：1.50.5 → 3 + (50-34)/2 = 3 + 8 = 3.18 ✓
static int InferBiocVersion(int pkgMinorVersion) {
    if (pkgMinorVersion >= 34) {
        return 3 + (pkgMinorVersion - 34) / 2;
    }
    return -1; // 无法推断
}

// 生成 Git Ref
// 格式：RELEASE_3_18
static std::wstring GenerateBiocRef(int biocVersionMajor, int biocVersionMinor) {
    return L"RELEASE_" + std::to_wstring(biocVersionMajor) + L"_" + std::to_wstring(biocVersionMinor);
}
